Instrucciones:

Dado un string compuesto de paréntesis (), llaves {}, y corchetes [], debes contar cuántos paréntesis correctamente emparejados hay — es decir, cuántos pares de ( y ) hay que estén balanceados y correctamente cerrados.

Proceso visual del conteo de paréntesis

Código Resuelto y explicado

function countParenthesis(arr) {
  const stack = []; // Creamos una pila (array) para guardar los paréntesis de apertura '('
  let count = 0; // Contador para llevar el total de pares de paréntesis válidos

  // Recorremos todo el arreglo de caracteres
  for (let i = 0; i < arr.length; i++) {
    const char = arr[i]; // Guardamos el carácter actual en una variable

    // Si el carácter es un paréntesis de apertura...
    if (char === '(') {
      stack.push(char); // Lo guardamos en la pila
    } else if (char === ')') { // Si es un paréntesis de cierre...
      if (stack.length > 0) { // Verificamos si hay algún '(' sin cerrar
        stack.pop(); // Quitamos el último '(' de la pila porque ya está emparejado
        count++; // Sumamos 1 al contador de pares válidos
      } // Si no hay '(', ignoramos ese ')' porque está "desbalanceado"
    }
  }

  return count; // Devolvemos el total de pares válidos encontrados
}

// Ejemplo de uso
const input = ['[', '(', '(', '(', '{', '(', ')', '}', ')', ']', ')'];
const result = countParenthesis(input);
console.log(result); // Salida: 3 — Hay 3 pares de paréntesis correctamente emparejados
